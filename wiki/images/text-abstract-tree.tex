\documentclass{minimal}
\usepackage[paperheight=1.55in,paperwidth=1.8in,
            height=1.55in,hoffset=0.05in,
	    voffset=0.05in,left=0in,width=1.8in]{geometry}
\usepackage{color}
\usepackage[usenames]{xcolor}
\usepackage{tikz}
\usepackage{scalefnt}
\usetikzlibrary{arrows}
\begin{document}
\raggedright
\begin{tikzpicture}[->,>=stealth',auto,
                    node distance=0.5in,semithick]
\tikzstyle{every node}=[draw,circle]
\node (A) {=};
\node (B) [below left of=A] {x};
\node (C) [below right of=A] {+};
\node (D) [below left of=C] {5};
\node (F) [below right of=C] {*};
\node (G) [below left of=F] {7};
\node (H) [below right of=F] {y};

\path (A) edge (B) edge (C)
      (C) edge (D) edge (F)
      (F) edge (G) edge (H);
\end{tikzpicture}
\end{document}
